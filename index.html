import axiosApi from 'utils/axios';
import { getAccountId } from 'api/auth';

export async function savePage({ id, data, lockVersion, prev, apply, rollback }) {
  sessionStorage.setItem('hideConfirmChangesForCurrentTab', '1'); // suppress self-broadcast
  apply(id, data); // optimistic UI
  try {
    const accountId = getAccountId();
    const res = await axiosApi.patch(`/acc/${accountId}/pages/${id}`, {
      ...data, lockVersion, currentStep: 'page',
    });
    return res.data;
  } catch (e) {
    if (e?.axiosResponse?.status === 409) throw new Error('Conflict: stale lock_version');
    rollback(id, prev);
    throw e;
  } finally {
    sessionStorage.removeItem('hideConfirmChangesForCurrentTab');
  }
}

Migration Strategy Overview

Phase 1: Nginx routing

Incoming request → Nginx

  • "api" subdomain → Rails (container: safariportal-web)
  • Everything else → React (container: safariportal-react)

The hybrid approach – Safari Portal architecture

Legacy Features (React SPA)

  • Itinerary Builder (complex drag-and-drop)
  • PDF rendering system
  • Existing user workflows

New Features (Rails + Hotwire)

  • Task Manager, Contact Management, Financials, Form Builder

Shared Infrastructure

  • Authentication (Rodauth), Common navigation

Phase 2: Single container

Incoming request → kamal-proxy (Traefik) → port 80 → Thruster

  • Static files served directly
  • Dynamic content → Puma (Rails)

container: safariportal-web

Where is React in this scheme?

React lives in static files

Thruster serves the built React app as static files.

Rails handles everything else dynamically.

One container. One deploy. Zero Nginx.

Shared Authentication

Shared Authentication

We've already been using Rodauth

  • Rodauth is modular and highly configurable
  • React app: Rodauth configuration with JWT-based session management
  • Rails app: new Rodauth configuration with traditional cookie-based session management

The question – how to share session between the apps?


class RodauthMain < RodauthBase
  include AccountRedirects

  configure do
    # ...

    login_redirect do
      if uses_two_factor_authentication?
        otp_auth_path
      else
        rails_routes.auth_jwt_path(
          redirect_uri: account_initial_path,
          format: :turbo_stream
        )
      end
    end
  end
end
					

class RodauthController < ApplicationController

  # ...

  def jwt
    rodauth_api = Rodauth::Rails.rodauth(:api, account: Current.user)
    access_token, refresh_token = rodauth_api.generate_jwt_tokens

    render turbo_stream: [
      turbo_stream.set_local_storage_item("accessToken", access_token),
      turbo_stream.set_local_storage_item("refreshToken", refresh_token),
      turbo_stream.redirect_to(params["redirect_uri"].presence || root_path)
    ]
  end
end

What if you need to embed Rails into React?

Rails views inside the React app

  • Itinerary builder → originally written in React
  • Book Button settings modal → Rails view rendered inside it
  • Tab navigation inside the modal → powered by Turbo

How?

With iframes

  • The React app renders an iframe as a container
  • The iframe content is loaded from Rails
  • The modal is rendered with the standard modal ViewComponent
  • A tiny Stimulus controller handles communication with React

class IframeModalController extends Controller {
  closeEvent(event) {
    window.parent.postMessage('closeModal', '*');
  }

  closeAfterSubmitEvent(event) {
    const submitter = event?.detail?.formSubmission?.submitter;
    const success = event?.detail?.success;

    if (!submitter) return;
    if (!submitter.dataset.ifameModalCloseOnClick) return;
    if (!success) return;

    this.closeEvent();
  }
}
